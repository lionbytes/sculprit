    function copyRamz(ramz){
      let clipboardTemp = document.createElement("textarea");
      document.body.appendChild(clipboardTemp);
      clipboardTemp.value = ramz;
      clipboardTemp.select();
      document.execCommand("copy");
      document.body.removeChild(clipboardTemp);

      copySuccessful();
    }

    // Show copy success message
    function copySuccessful() {
      $('.copysuccess').show().animate( { top: '-25px' } , 500 ).fadeOut(250);
      setTimeout(function(){ $('.copysuccess').removeAttr('style'); }, 750);
    }

    // Closing Auto-Copy Modal
    const autocopyModal = document.querySelector("#autocopyModal");
    function closeModal() {
      autocopyModal.classList.add('d-none') ;
    }
    document.addEventListener('keydown', (event) => {
      if (event.key == 'Escape' && !autocopyModal.classList.contains('d-none')) { 
        closeModal();
      }
    });
    
    let modalEscape = document.querySelector('#autocopyModal');
      modalEscape.addEventListener( 'click', (event) => {
          if (event.target.id === 'autocopyModal'){
            closeModal();
          }
      });
      
  </script>
